                $("#btnSunset").on("click", function () {

                let bldg1 = parseInt($("#sunsetIn1").val());
                let bldg2 = parseInt($("#sunsetIn2").val());
                let bldg3 = parseInt($("#sunsetIn3").val());
                let bldg4 = parseInt($("#sunsetIn4").val());
                let bldg5 = parseInt($("#sunsetIn5").val());
                let bldg6 = parseInt($("#sunsetIn6").val());

                let bldgArray = [bldg1, bldg2, bldg3, bldg4, bldg5, bldg6];

                //let output = [bldg1]
                let output = new Array();
                let tallest = 0;
                let currentBldg = 0;
                //here is where to determine if current is larger or smaller than the one before it
                for (let loop = 0; loop <= 5; loop++) {
                currentBldg = bldgArray[loop];
                if (currentBldg > tallest) {
                output.push(currentBldg);
                tallest = currentBldg;

                }



                //show output
                $("#sunsetOut").html(`Given the input, ${output.length} building(s) can see the sunset. 
 The buildings that can see the sunset are [${output.toString()}]`)



                }

                })
